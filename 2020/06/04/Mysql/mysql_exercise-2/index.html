– 获取一周的日子

SET @bd = DAYOFWEEK( @dt );

SET @startmonth = MONTH (
DATE_ADD( @dt, INTERVAL - @bd + 1 DAY ));

SET @startday = DAY (
DATE_ADD( @dt, INTERVAL - @bd + 1 DAY ));

SET @endmonth = MONTH (
DATE_ADD( @dt, INTERVAL 7 - @bd DAY ));

SET @endday = DAY (
DATE_ADD( @dt, INTERVAL 7 - @bd DAY ));
SELECT
Student.*
FROM
Student
WHERE
MONTH ( s_birth ) >= @startmonth
AND MONTH ( s_birth ) <= @endmonth AND DAY ( s_birth ) >= @startday
AND DAY ( s_birth ) <= @endday
